CString GetMacAddress()
{
	PIP_ADAPTER_ADDRESSES pIpAddress;
	ULONG size = 0;
	DWORD retval;
	CString strRet = _T("");
	
	retval = GetAdaptersAddresses(AF_INET, 0, 0, NULL, &size);
	
	if (retval != ERROR_BUFFER_OVERFLOW)
		return strRet;
	pIpAddress = new IP_ADAPTER_ADDRESSES[ (size / sizeof(IP_ADAPTER_ADDRESSES) + 1)];
	retval = GetAdaptersAddresses(AF_INET, 0 ,NULL,pIpAddress,&size);
	if (retval != 0)
		return strRet;

	IP_ADAPTER_ADDRESSES * pTemp = pIpAddress;
	BYTE *bt_mac;
	CString strSubnet;
	while (pTemp)
	{
		if (pTemp->FirstUnicastAddress)
		{	
			bt_mac = pTemp->PhysicalAddress;
			strRet.Format(_T("%.2X-%.2X-%.2X-%.2X-%.2X-%.2X"), bt_mac[0], bt_mac[1], bt_mac[2], bt_mac[3], bt_mac[4], bt_mac[5]);
			break;
		}
		pTemp= pTemp->Next;
	}
	delete []pIpAddress;
	return strRet;
}
